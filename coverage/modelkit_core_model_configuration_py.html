45def walk_objects(mod): 

46 already_seen = set() 

47 try: 

48 for _, modname, _ in pkgutil.walk_packages(mod.__path__, mod.__name__ + "."): 

49 submod = importlib.import_module(modname) 

50 yield from walk_module_objects(submod, already_seen)

55def _configurations_from_objects(m) -> Dict[str, ModelConfiguration]: 

56 if inspect.isclass(m) and issubclass(m, Asset): 

57 return { 

58 key: ModelConfiguration(**{**config, "model_type": m}) 

59 for key, config in m.CONFIGURATIONS.items() 

60 } 

61 elif isinstance(m, (list, tuple)): 

62 return dict(ChainMap(*(_configurations_from_objects(sub_m) for sub_m in m))) 

63 elif isinstance(m, ModuleType): 

64 return { 

65 key: ModelConfiguration(**{**config, "model_type": m})

67 for key, config in m.CONFIGURATIONS.items()

69 elif isinstance(m, str): 

70 models = [importlib.import_module(modname) for modname in m.split(",")]
